class CustomStep(SimpleStep):

    name = Property.Number(label="Test")

    i = 0
    
    @action(key="name", parameters=None)
    def test(self, **kwargs):
        self.name="WOOHOO"

    async def run_cycle(self):

        print("RUN STEP", self.id, self.name, self.__dict__)
